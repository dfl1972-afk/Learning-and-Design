                        card.classList.add('project-highlight');

                    }

                });

            }

                const relatedSkills = projectsData[projectId].skills;

                    if (relatedSkills.includes(card.dataset.skill)) {

                        card.classList.add('skill-highlight');

                    }

                });

            }
